var createScene = function () { var scene = new BABYLON.Scene(engine); scene.clearColor = new BABYLON.Color3(0, 0, 0); // Matahari (core sphere) var coreSphere = BABYLON.MeshBuilder.CreateSphere("coreSphere", {diameter: 2.01, segments: 64}, scene); coreSphere.position = new BABYLON.Vector3(0, 5, -10); var coreMat = new BABYLON.StandardMaterial("coreMat", scene); coreMat.emissiveColor = new BABYLON.Color3(1, 0.6, 0.1); coreSphere.material = coreMat; // Ground var ground = BABYLON.MeshBuilder.CreateGround("ground", {width: 40, height: 40}, scene); ground.position.y = -2; var groundMat = new BABYLON.StandardMaterial("groundMat", scene); groundMat.diffuseColor = new BABYLON.Color3(0.3, 0.3, 0.3); ground.material = groundMat; ground.receiveShadows = true; // Objek var sphere = BABYLON.MeshBuilder.CreateSphere("sphere", {diameter: 1}, scene); sphere.position.y = -1.5; sphere.position.x = -2; var box = BABYLON.MeshBuilder.CreateBox("box", {size: 1.2}, scene); box.position.y = -1.4; box.position.x = 2; var torus = BABYLON.MeshBuilder.CreateTorus("torus", {diameter: 1.5, thickness: 0.3}, scene); torus.position.y = -1.3; torus.position.z = 3; // Cahaya dari matahari var light = new BABYLON.DirectionalLight("sunLight", new BABYLON.Vector3(0, -1, 0), scene); light.position = coreSphere.position.clone(); var shadowGen = new BABYLON.ShadowGenerator(1024, light); shadowGen.useExponentialShadowMap = true; [sphere, box, torus, coreSphere].forEach(mesh => { shadowGen.addShadowCaster(mesh); }); // Update posisi & arah cahaya scene.registerBeforeRender(function () { let t = Date.now() * 0.001; let radius = 10; coreSphere.position.x = Math.sin(t) * radius; coreSphere.position.z = Math.cos(t) * radius; coreSphere.position.y = 5; light.position = coreSphere.position.clone(); var target = new BABYLON.Vector3(0, -1, 0); light.direction = target.subtract(coreSphere.position).normalize(); }); // Interaksi: klik objek = animasi scene.onPointerObservable.add((pointerInfo) => { if (pointerInfo.type === BABYLON.PointerEventTypes.POINTERPICK) { let pickedMesh = pointerInfo.pickInfo.pickedMesh; if (pickedMesh === sphere) { let anim = new BABYLON.Animation("bounce", "position.y", 30, BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CYCLE); let keys = [ { frame: 0, value: sphere.position.y }, { frame: 10, value: sphere.position.y + 2 }, { frame: 20, value: sphere.position.y } ]; anim.setKeys(keys); sphere.animations = [anim]; scene.beginAnimation(sphere, 0, 20, false); } if (pickedMesh === box) { let anim = new BABYLON.Animation("rotateBox", "rotation.y", 30, BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CYCLE); let keys = [ { frame: 0, value: box.rotation.y }, { frame: 30, value: box.rotation.y + Math.PI * 2 } ]; anim.setKeys(keys); box.animations = [anim]; scene.beginAnimation(box, 0, 30, false); } if (pickedMesh === torus) { let anim = new BABYLON.Animation("slideTorus", "position.x", 30, BABYLON.Animation.ANIMATIONTYPE_FLOAT, BABYLON.Animation.ANIMATIONLOOPMODE_CYCLE); let keys = [ { frame: 0, value: torus.position.x }, { frame: 15, value: torus.position.x + 2 }, { frame: 30, value: torus.position.x } ]; anim.setKeys(keys); torus.animations = [anim]; scene.beginAnimation(torus, 0, 30, false); } } }); scene.createDefaultCamera(true, undefined, true); scene.activeCamera.lowerRadiusLimit = 2; return scene; };
